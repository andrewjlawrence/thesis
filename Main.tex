\newcommand{\contact}[1]{
  %% Draw a -
  -- ++(1,0)
  %% Draw ]
  +(-0.1, 0.5) -- +(0, 0.5) -- +(0, -0.5)  -- +(-0.1, -0.5)
  %% Move across
  ++(0.4, 0)
  %% Draw text above
  +(0,0.8) node{#1}
  %% Move across
  ++(0.4, 0)
  %% Draw [
  +(0.1, 0.5) -- +(0, 0.5) -- +(0, -0.5)  -- +(0.1, -0.5)
  %% Reset to current point
  ++(0, 0) --
  %% Draw a -
  ++(1,0) 
}



\newcommand{\closedContact}[1]{
  %% Draw a -
  -- ++(1,0)
  %% Draw ]
  +(-0.1, 0.5) -- +(0, 0.5) -- +(0, -0.5)  -- +(-0.1, -0.5)
  %% Move across
  ++(0.4, 0)
  %% Draw text above
  +(0,0.8) node{#1}
  %% Draw close i.e. /
  +(-0.2, -0.5) -- +(0.2, 0.5)
  %% Reset to current point (i.e. middle)
  ++(0, 0)
  %% Move across
  ++(0.4, 0)
  %% Draw [
  +(0.1, 0.5) -- +(0, 0.5) -- +(0, -0.5)  -- +(0.1, -0.5)
  %% Reset to current point
  ++(0, 0) --
  %% Draw a -
  ++(1,0) 
}

\newcommand{\coil}[1]{
   %% Draw a -
  -- ++(1,0)
  %% Move to end of -
  +(0.5,-0.5)
  %% Draw arc (
  arc (270:90:0.5cm)
  %% Move across
  ++(0.3, 0)
  %% Draw text above
  +(0,0.3) node{#1}
  %% Reset
  +(0.3,-1)
  %% Draw Arc )
  arc (-90:90:0.5cm)
  %% Move to center of )
  ++(0.5,-0.5)
  %% Finish Path
  ;
}
